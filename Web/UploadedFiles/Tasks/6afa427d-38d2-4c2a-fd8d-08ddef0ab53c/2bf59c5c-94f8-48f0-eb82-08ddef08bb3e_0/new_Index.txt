        $.AjaxJsonRequest(_url, _data, function (reponse) {

                reponse.lst = [];
            }
            $("#tblSection").dataTable({
                "bProcessing": true,
                "aaData": reponse.lst,
                "deferRender": true,
                "destroy": true,
                "paging": false,
                "scroller": true,
                "scrollY": 200,
                "aoColumns": [
                       // { "sTitle": "Order Id", "mData": "LabOrderId" },

                            }
                        }


                ]
            });
            $("#All").prop('checked', true);
            $("#All").trigger('change');



        });
    }

    function SetlabSection(check, id) {
        try {
            if (check) {
                debugger;
                arrlabSections.push(id);
            }
            else {
                if (arrlabSections != null) {
                    for (var i = 0; i < arrlabSections.length; i++) {
                        if (arrlabSections[i] == id) {
                            arrlabSections.splice(i, 1);
                            break;
                        }

                    }
                }
            }

        } catch (e) {
            console.log(e.message);
        }
    }

    function GetPartyDetailsByPartyId(partyId) {
        var _url = '@Url.Action("GetPartyDetailsByPartyId", "Report")';
        var _data = "{partyId:'" + partyId + "'}";
        $.AjaxJsonRequest(_url, _data, function (msg) {
            if (msg != null) {
                var items = '';
                $("#ddl_PartyLocation").empty();
                if (msg.length > 0) {
                    $.each(msg[0], function (index, item) {

                        items += "<option value='" + item.PartyLocationId + "'>" + item.PartyLocationName + "</option>";
                    });
                    $("#ddl_PartyLocation").html(items);
                }
            }
            else {
                ShowMessage('1', "NO Record Found");
            }
        });
    }
