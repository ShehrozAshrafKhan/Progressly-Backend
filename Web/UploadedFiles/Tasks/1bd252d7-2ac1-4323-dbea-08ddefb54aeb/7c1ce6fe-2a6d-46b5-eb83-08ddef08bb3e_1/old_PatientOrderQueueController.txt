                    if (!string.IsNullOrEmpty(WardId))
                        lpm.Add(new ParameterList("WardId", WardId));
                    if (!string.IsNullOrEmpty(IpNo))
                        lpm.Add(new ParameterList("IPNo", IpNo));
                }

                dt = new PatientAdmitRepository().GetAllPatientAdmitData(lpm);
                if (dt != null)
                {
                    lstPatAdmit = (from item in dt.AsEnumerable()
                                   select new
                                   {
                                       PatientAdmitId = item.Field<long>("PatientAdmitId"),
                                       PatientId = item.Field<long>("PatientId"),
                                       PatientName = item.Field<string>("PatientName"),
                                       IPNo = item.Field<string>("IPNo"),
                                       PatientNo = item.Field<string>("PatientNo"),
                                       ComponentName = item.Field<string>("ComponentName"),
                                       WardName = item.Field<string>("WardName"),
                                       BedNo = item.Field<string>("BedNo"),
                                       Mobile1 = item.Field<string>("Mobile1"),
                                       Consultant = item.Field<string>("InternalConsultant"),
                                       IsDischarge = item.Field<bool>("IsDischarge"),
                                       AgeGender = (item.Field<DateTime?>("DOB") == null ? item.Field<string>("AgeGender") : item.Field<string>("PatientAge")) + " / " + item.Field<string>("Gender").ToString(),
                                       AdmitDate = item.Field<DateTime>("AdmitDate"),
                                       AdvanceAmount = item.Field<decimal>("AdvanceAmount"),
                                       AdmissionFee = item.Field<decimal>("AdmissionFee"),
                                       PatientOrderSetId = item.Field<long?>("PatientOrderSetId"),
                                       Print = "<a href='#' onclick=\"OpenPrintPreview('/HMS/Report/PrintPatientAdmissionOrder?Id=" + item.Field<long>("PatientId") + "')\"><img src='/Images/Common/Printer.png' style='width:30px;height:30px;' /></a>",
                                       Proceed = (item.Field<bool>("IsDischarge") == false ? "<a class='ProceedIcon' href='/LIMS/LabOrder/ProceedFromAdmissionQueue?Id=" + item.Field<long>("PatientAdmitId") + "&Department=" + Department + "'></a>" : ""),
                                   }).ToList();
                }
            }
            catch (Exception ex)
            {
                return objJSS.Serialize(new { lstPatAdmit, MessageText = ex.Message });
            }
            return objJSS.Serialize(new { lstPatAdmit, MessageText = "" });
        }

                List<ParameterList> lpm = new List<ParameterList>();

                if (department != "")
                    lpm.Add(new ParameterList("DepartmentShortName", department));
                lpm.Add(new ParameterList("LocationId", BaseRepository.GetLoginLocationId()));

                if (category != "$")
                    lpm.Add(new ParameterList("PatientCategoryProfileId", long.Parse(category)));
                if (orderNo != "")
                    lpm.Add(new ParameterList("PatientOrderNo", orderNo));

                if (Convert.ToBoolean(cbDate))
                {
                    DateTime toDateTime = Convert.ToDateTime(toDate);
                    DateTime fromDateTime = Convert.ToDateTime(fromDate);
                    lpm.Add(new ParameterList("ToDate", toDateTime));
                    lpm.Add(new ParameterList("FromDate", fromDateTime));
                }
                if (!string.IsNullOrEmpty(MobileNo))
                    lpm.Add(new ParameterList("Mobile", MobileNo));
                DataTable dt = null;
                dt = CommonRepository.GetDataTable("OutstandingFORInquiry", "[hms].[procGetPatientOrderForLIMS]", lpm);
                if (dt != null)
                {
                    orderList = (from item in dt.AsEnumerable()
                                 select new
                                 {
                                     PatientInquiryId = item.Field<long>("PatientInquiryId"),
                                     PatientOrderOn = item.Field<DateTime>("PatientOrderOn").ToString("dd/MM/yyyy"),
                                     PatientNo = item.Field<string>("PatientNo"),
                                     PatientId = item.Field<long>("PatientId"),
                                     PatientName = item.Field<string>("PatientName").ToString(),
                                     Age = (item.Field<DateTime?>("DOB") == null ? "" : item.Field<string>("PatientAge")) + " " + item.Field<string>("ProfileName").ToString(),
                                     Mobile = item.Field<string>("Mobile1"),

                                     //Print = "<a class='PrintIcon' href='/LIMS/Report/PrintPatientInvoice?Id=" + item.Field<long>("PatientInquiryId") + "' target='_blank'></a>",
