        public ActionResult SaveRecord(PatientOrder obj)
        {
            JsonResponseResult res = new JsonResponseResult();

            try
            {

                object returnId = new PatientOrderRepository().Save(ref obj);
                CommonRepository.GetDataForUserAvtivity<PatientOrder>(obj, ref objUserActivityLog);

                //if (Convert.ToInt64(returnId) <= 0)
                //    res.Id = returnId;
                //else
                res.Id = obj.PatientOrderId;


                res.msgObj = new MessageObject(MessageTypeMode.SAVED);

                bool isa4 = BaseRepository.GetOrginizationSetting.IsA4PatientInvoice;
                if (isa4)
                {
                    res.isa4 = isa4;
                }
                else
                {
                    res.isa4 = false;
                }
            }

            catch (Exception ex)
            {
                res.msgObj = new MessageObject(MessageTypeMode.ERROR, ex.Message);
                return Json(res, JsonRequestBehavior.AllowGet);
            }
            return Json(res, JsonRequestBehavior.AllowGet);
        }

        public ActionResult SaveRecordInsurance(List<PatientOrder> listObject)
        {
            List<JsonResponseResult> listres = new List<JsonResponseResult>();

            try
            {
                foreach (var Item in listObject)
                {
                    var obj = Item;
                    object returnId = new PatientOrderRepository().Save(ref obj);
                    CommonRepository.GetDataForUserAvtivity<PatientOrder>(obj, ref objUserActivityLog);


                    JsonResponseResult res = new JsonResponseResult();
                    //if (Convert.ToInt64(returnId) <= 0)
                    //    res.Id = returnId;
                    //else
                    res.Id = obj.PatientOrderId;


                    res.msgObj = new MessageObject(MessageTypeMode.SAVED);

                    bool isa4 = BaseRepository.GetOrginizationSetting.IsA4PatientInvoice;
                    if (isa4)
                    {
                        res.isa4 = isa4;
                    }
                    else
                    {
                        res.isa4 = false;
                    }

                    listres.Add(res);
                }

            }

            catch (Exception ex)
            {
                foreach (var res in listres)
                {
                    res.msgObj = new MessageObject(MessageTypeMode.ERROR, ex.Message);
                    return Json(res, JsonRequestBehavior.AllowGet);
                }

            }
            return Json(listres, JsonRequestBehavior.AllowGet);
        }

        public string GetLabOrderData(string Id)
        {
            var serviceList = new List<object>();
            var objJSS = new JavaScriptSerializer() { MaxJsonLength = Int32.MaxValue };
            List<Dictionary<string, string>> lstDic = new List<Dictionary<string, string>>();
            try
            {
                var objPatientOrder = new PatientOrderRepository().GetById(Id, false);
                if (objPatientOrder != null)
                {
                    objPatientOrder.Patient = new PatientRepository().GetById(objPatientOrder.PatientId, false);
                    objPatientOrder.PatientOrderSet = new PatientOrderSetRepository().GetById(objPatientOrder.PatientOrderSetId, false);
                    objPatientOrder.Patient.Profile = new ProfileRepository().GetById(objPatientOrder.PatientOrderSet.BillCategoryProfileId, false);
                }
                List<ParameterList> lpm = new List<ParameterList>();
                lpm.Add(new ParameterList("PatientOrderId", Id));
                lpm.Add(new ParameterList("LocationId", BaseRepository.GetLoginLocationId()));
                lpm.Add(new ParameterList("OrginizationId", BaseRepository.GetLoginOrginizationId()));
                if (objPatientOrder.PatientOrderSet != null && objPatientOrder.PatientOrderSet.DepartmentComponentId != null)
                {
                    var objDepartmentComponent = new DepartmentComponentRepository().GetById(objPatientOrder.PatientOrderSet.DepartmentComponentId, false);
                    lpm.Add(new ParameterList("DepartmentGroupId", objDepartmentComponent.DepartmentGroupId));
                }
                DataTable dt;

                dt = CommonRepository.GetDataTable("GetByBillCategoryProfileIdForProcedureForEdit", "[hms].[procGetPatientOrderForProcedure]", lpm);


                if (dt != null)
                {
                    for (int i = 0; i < dt.Rows.Count; i++)
                    {
                        Dictionary<string, string> dic = new Dictionary<string, string>();
                        for (int j = 0; j < dt.Columns.Count; j++)
                        {
                            dic.Add(dt.Columns[j].ColumnName, dt.Rows[i][j].ToString());
                        }
                        lstDic.Add(dic);
                    }

                }
            }
            catch (Exception ex)
            {
                return string.Empty;
            }

            return objJSS.Serialize(lstDic);
        }

        public JsonResult SearchPatientById(string patientId)

                if (objPatient != null)
                {
                    objPerson = null;
                    if (objPatient.IsFinancialAssistanceRequired)

                        fiAssistance = new FinancialAssistanceRepository().GetByPatientId(objPatient.PatientId, false);

        public ActionResult GetShiftRecord(string OrderOn)
        {
            ShiftRecord objShiftRecord = null;
            string floorSchedule = "";
            try
            {
                bool isshowFloorInfo = BaseRepository.GetLocationSetting.IsShowFloorInfoOnReception;
                if (isshowFloorInfo && BaseRepository.GetLocationSetting != null && BaseRepository.GetLocationSetting.IsAllowShiftFlow)
                {
                    objShiftRecord = new ShiftRecordRepository().GetByDateAndUserId(BaseRepository.GetLoginUserId(), OrderOn, false);
                    if (objShiftRecord != null)
                    {
                        objShiftRecord.ShiftLocation = new ShiftLocationRepository().GetById(objShiftRecord.ShiftLocationId, false);
                        List<dynamic> FloorInformationList = new List<dynamic>();
                        var objJSS = new JavaScriptSerializer() { MaxJsonLength = int.MaxValue };
                        List<ParameterList> lpm = new List<ParameterList>();
                        lpm.Add(new ParameterList("LocationId", BaseRepository.GetLoginLocationId()));
                        lpm.Add(new ParameterList("ShiftId", objShiftRecord.ShiftLocation.ShiftId));
                        lpm.Add(new ParameterList("Date", Convert.ToDateTime(OrderOn)));




                        DataTable dt = CommonRepository.GetDataTable("BlookFloorList", "hms.procGetComponentStatus", lpm);

                        if (dt != null && dt.Rows.Count > 0)
                        {
                            FloorInformationList = (from item in dt.AsEnumerable()
                                                    select new
                                                    {
                                                        BlockFloorId = item.Field<long>("BlockFloorId"),
                                                        ScheduleTypeProfileId = item.Field<int>("ScheduleTypeProfileId"),
                                                        FloorNo = item.Field<string>("FloorNo"),
                                                        Type = item.Field<string>("ScheduleType"),
                                                        TotalBooking = item.Field<int>("TotalBooking"),
                                                        TotalBooked = item.Field<int>("TotalBooked"),
                                                        ReportProceeded = item.Field<int>("TotalReportProceeded"),
                                                        Load = item.Field<int>("TotalBooked") - item.Field<int>("TotalReportProceeded"),
                                                        TotalFree = item.Field<int>("TotalFree"),
                                                    }).ToList<dynamic>();

                            FloorInformationList = FloorInformationList.OrderByDescending(x => x.Type).OrderBy(x => x.FloorNo).ToList();

                            floorSchedule = objJSS.Serialize(FloorInformationList);
                        }
                    }
                }
            }
            catch (Exception ex)
            {
                return Json(new MessageObject(MessageTypeMode.ERROR, ex.Message), JsonRequestBehavior.AllowGet);
            }
            return this.Json(new { obj = objShiftRecord, floorSchedule = floorSchedule, LocationSetting = BaseRepository.GetLocationSetting }, JsonRequestBehavior.AllowGet);
        }

                var PaymentModePrefix = new ProfileRepository().GetById(obj.PatientOrderSet.PaymentMethodProfileId).ProfilePrefix;
                if (PaymentModePrefix == ProfilePrefixMode.PATPMD.ToString())
                {
                    patientAccount = new PatientRepository().GetPatientAdvanceByPatientNo(obj.PatientId, false);
                }
            }
            obj.Patient = new PatientRepository().GetById(obj.PatientId, false);
            obj.ParentPatientOrderId = Convert.ToInt64(Id);
            obj.PatientOrderOn = DateTime.Now;
            obj.PatientOrderId = 0;
            TempData["department"] = ShortName;
            TempData["ApprovedInsuranceProcedure"] = obj;
            TempData["patientAccount"] = patientAccount;
            return RedirectToAction("InsuranceIndex");
        }
        public string GetDentalData(string Id, string shortName)
        {
            var objJSS = new JavaScriptSerializer() { MaxJsonLength = Int32.MaxValue };
            List<Dictionary<string, string>> lstDic = new List<Dictionary<string, string>>();
            try
            {

                int? departmentGroupId = null;
                var obj = new PatientOrderRepository().GetById(Id, false);
                obj.PatientOrderSet = new PatientOrderSetRepository().GetById(obj.PatientOrderSetId, false);
                if (obj.PatientOrderSet != null && obj.PatientOrderSet.DepartmentComponentId != null)
                {
                    var objDepartmentComponent = new DepartmentComponentRepository().GetById(obj.PatientOrderSet.DepartmentComponentId, false);
                    departmentGroupId = objDepartmentComponent.DepartmentGroupId;
                }

                List<ParameterList> lpm = new List<ParameterList>();
                lpm.Add(new ParameterList("PatientOrderId", Id));
                if (!string.IsNullOrEmpty(shortName))
                    lpm.Add(new ParameterList("DepartmentShortName", shortName));
                if (departmentGroupId != null)
                    lpm.Add(new ParameterList("DepartmentGroupId", departmentGroupId));
                DataTable dt = CommonRepository.GetDataTable("GetForDental", "[hms].[procGetPatientOrderForProcedure]", lpm);
                if (dt != null)
                {
                    for (int i = 0; i < dt.Rows.Count; i++)
                    {

                        Dictionary<string, string> dic = new Dictionary<string, string>();
                        for (int j = 0; j < dt.Columns.Count; j++)
                        {
                            dic.Add(dt.Columns[j].ColumnName, dt.Rows[i][j].ToString());
                        }
                        lstDic.Add(dic);
                    }
                }
            }
            catch (Exception)
            {
                return string.Empty;
            }

            return objJSS.Serialize(lstDic);
        }

        public string GetAllClinicPatientOrder(string department, string category, string orderNo, string MobileNo, string QID, string MRN,string VisitNo, string fromDate, string toDate, string cbDate)
        {
            var listFinal = new List<object>();
            var lstPatientInvoices = (object)null;
            var orderQueueList = (object)null;



            var objJSS = new JavaScriptSerializer() { MaxJsonLength = Int32.MaxValue };

            try
            {
                List<ParameterList> lpm = new List<ParameterList>();

                var id = new DocMovementRepository().GetByShortName("CLNPAP", false).DocMovementId;
                lpm.Add(new ParameterList("DocMovementId", id));
                if (department != "")
                    lpm.Add(new ParameterList("DepartmentShortName", department));

                lpm.Add(new ParameterList("LocationId", BaseRepository.GetLoginLocationId()));
                if (category != "$")
                {
                    lpm.Add(new ParameterList("PatientCategoryProfileId", long.Parse(category)));
                }
                if (orderNo != "")
                {
                    lpm.Add(new ParameterList("PatientOrderNo", orderNo));
                }
                if (MobileNo != "")
                {
                    lpm.Add(new ParameterList("MobileNo", MobileNo));
                }
                if (QID != "")
                {
                    lpm.Add(new ParameterList("CNIC", QID));
                }
                if (MRN != "")
                {
                    lpm.Add(new ParameterList("PatientNo", MRN));
                } 
                if (VisitNo != "")
                {
                    lpm.Add(new ParameterList("PatientOrderNo", VisitNo));
                }
                if (Convert.ToBoolean(cbDate))
                {
                    DateTime toDateTime = Convert.ToDateTime(toDate);
                    DateTime fromDateTime = Convert.ToDateTime(fromDate);
                    lpm.Add(new ParameterList("ToDate", toDateTime));
                    lpm.Add(new ParameterList("FromDate", fromDateTime));
                }

                DataTable dtQueue = null;
                DataTable dtPatientInvoice = null;

                dtQueue = CommonRepository.GetDataTable("OPDPatientPrescribeFORProcedures", "[hms].[procGetPatientOrderForLIMS]", lpm);
                lpm.RemoveAll(p => p.ParameterName == "DocMovementId");
                id = new DocMovementRepository().GetByShortName("PROPOD", false).DocMovementId;
                lpm.Add(new ParameterList("DocMovementId", id));
                dtPatientInvoice = CommonRepository.GetDataTable("LISTForProcedureInovice", "[hms].[procGetPatientOrder]", lpm);

                string rolePrefix = "";
                if (BaseRepository.CheckLoginUser != null && BaseRepository.CheckLoginUser.Role != null)
                    rolePrefix = BaseRepository.CheckLoginUser.Role.RolePrefix;

                if (dtQueue != null)
                {
                    orderQueueList = (from item in dtQueue.AsEnumerable()
                                      select new
                                      {
                                          LabOrderId = item.Field<long>("PatientOrderId"),
                                          PatientOrderNo = item.Field<string>("PatientOrderNo"),
                                          PatientOrderOn = item.Field<DateTime>("PatientOrderOn").ToString("dd/MM/yyyy"),
                                          PatientNo = item.Field<string>("PatientNo"),
                                          PatientId = item.Field<long>("PatientId"),
                                          PatientName = item.Field<string>("PatientName").ToString(),
                                          Age = (item.Field<DateTime?>("DOB") == null ? "" : item.Field<string>("PatientAge")) + " " + item.Field<string>("ProfileName").ToString(),
                                          Mobile = item.Field<string>("Mobile1"),
                                          TestName = item.Field<string>("Tests"),
                                          TotalAmount = item.Field<long>("TotalAmount"),
                                          TotalDiscount = item.Field<decimal?>("TotalDiscount"),
                                          ComponentName = item.Field<string>("ComponentName"),
                                          InsuranceCompanyName = item.Field<string>("InsuranceCompanyName"),
                                          ProcedureName = item.Field<string>("ProcedureName"),
                                          InsuranceStatus = item.Field<string>("InsuranceStatus"),
                                          InsuranceStatusFilePath = item.Field<string>("InsuranceStatusFilePath"),
                                          BillCategoryName = (item.Field<string>("BillCategoryName") == "Insurance" ? item.Field<string>("InsuranceCompanyName") : item.Field<string>("BillCategoryName")),
                                          Proceed = (item.Field<string>("BillCategoryName") != "Insurance" ? "<a class='ProceedIcon' href='/eClinic/ProcedureInvoice/ProceedDental?Id=" + item.Field<long>("PatientOrderId") + "&&ShortName=" + department + "'></a>" : "<a class='ProceedIcon' href='/eClinic/ProcedureInvoice/ProceedInsurance?Id=" + item.Field<long>("PatientOrderId") + "&&ShortName=" + department + "'></a>"),
                                      }).ToList();
                }
                listFinal.Add(orderQueueList);
                if (dtPatientInvoice != null)
                {
                    lstPatientInvoices = (from item in dtPatientInvoice.AsEnumerable()
                                          select new
                                          {
                                              MRN = item.Field<string>("MRN"),
                                              PatientName = item.Field<string>("PatientName").ToString(),
                                              PatientOrderNo = item.Field<string>("PatientOrderNo"),
                                              InvoiceNo = item.Field<string>("InvoiceNo"),
                                              Age = (item.Field<DateTime?>("DOB") == null ? "" : item.Field<string>("PatientAge")) + " " + item.Field<string>("ProfileName").ToString(),
                                              BillCategoryName = (item.Field<string>("BillCategoryName") == "Insurance" ? item.Field<string>("InsuranceCompanyName") : item.Field<string>("BillCategoryName")),
                                              InsuranceCompanyName = item.Field<string>("InsuranceCompanyName"),
                                              ProcedureName = item.Field<string>("ProcedureName"),
                                              ComponentName = item.Field<string>("ComponentName"),
                                              PatientOrderOn = item.Field<DateTime>("PatientOrderOn").ToString("dd/MM/yyyy"),
                                              Print = "<a href='#' onclick=PrintPatientSlip('" + item.Field<long>("PatientOrderId") + "');><img src='/Images/Common/Printer.png' style='width:30px;height:30px;' /></a>",

                                              LabOrderId = item.Field<long>("PatientOrderId"),
                                              Payment = item.Field<decimal?>("TotalReceived") ?? 0,
                                              PatientId = item.Field<long>("PatientId"),
                                              Mobile = item.Field<string>("Mobile1"),
                                              TotalAmount = item.Field<long>("TotalAmount"),
                                              Edit = IsEditEnable(rolePrefix) == true ? "<a class='EditIcon' href='/eClinic/ProcedureInvoice/Edit/" + item.Field<long>("PatientOrderId") + "'></a>" : "",

                                          }).ToList();
                }

                listFinal.Add(lstPatientInvoices);
            }
            catch (Exception ex)
            {
                return ex.ToString();
            }
            return objJSS.Serialize(listFinal);
        }
