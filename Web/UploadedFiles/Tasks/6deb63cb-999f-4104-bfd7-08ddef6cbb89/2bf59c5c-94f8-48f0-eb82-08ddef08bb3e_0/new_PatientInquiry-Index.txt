    function splitArrayInHalf(arr) {
        const thresholdHour = 16; // 4:00 PM
        const thresholdMinute = 0; // 00 minutes

        const firstHalf = [];
        const secondHalf = [];

        arr.forEach(row => {
            const time = new Date(parseInt(row.StartDateTime.substr(6)));
            const rowHour = time.getHours();
            const rowMinute = time.getMinutes();

            if (rowHour < thresholdHour || (rowHour === thresholdHour && rowMinute < thresholdMinute)) {
                firstHalf.push(row);
            } else {
                secondHalf.push(row);
            }
        });

        return [firstHalf, secondHalf];
    }

    function CalculateAge() {
      var  dateString = $("#dpkDOB").getValue();
        var parts = dateString.split('/');
        var day = parseInt(parts[0], 10);
        var month = parseInt(parts[1], 10) - 1; // Subtract 1 because months are zero-indexed in JavaScript
        var year = parseInt(parts[2], 10);
        var dob = new Date(year, month, day);
        if (!CheckIsValidDate(dob) || dob == null || dob == '') {
            $("#txtYear").setValue('0');
            $("#txtMonth").setValue('0');
            $("#txtDay").setValue('0');
            $("#txtHour").setValue("0");
            return;
        }
        var now = new Date();
        var today = new Date(now.getYear(), now.getMonth(), now.getDate());

        var yearNow = now.getYear();
        var monthNow = now.getMonth();
        var dateNow = now.getDate();

        var yearDob = dob.getYear();
        var monthDob = dob.getMonth();
        var dateDob = dob.getDate();

        yearAge = yearNow - yearDob;

        if (monthNow >= monthDob)
            var monthAge = monthNow - monthDob;
        else {
            yearAge--;
            var monthAge = 12 + monthNow - monthDob;
        }

        if (dateNow >= dateDob)
            var dateAge = dateNow - dateDob;
        else {
            monthAge--;
            if (monthDob > 2)
                var dateAge = 31 + dateNow - dateDob;
            else
                var dateAge = 28 + dateNow - dateDob;

            if (monthAge < 0) {
                monthAge = 11;
                yearAge--;
            }
        }
        var hours = 0;
        //var patAge = selectedPatientAge.split(' ');
        //if (patAge.length > 1) {
        //    if (patAge[1] == "Hour(s)")
        //        hours = patAge[0]
        //}

        if (yearAge > 200) {
            ShowMessage('1', MSG_AGE);
            $("#txtPatientAge").setValue('');
            $("#txtYear").focus();
            return;
        }
        else if (dateAge <= 0 && hours == 0) {
            if (yearAge == 0 && monthAge == 0) {
                ShowMessage('1', MSG_AGE);
                $("#txtPatientAge").setValue('');
                $("#txtYear").focus();
                return;
            }
        }
        $("#txtYear").setValue(yearAge);
        $("#txtMonth").setValue(monthAge);
        $("#txtDay").setValue(dateAge);
        $("#txtHour").setValue("0");

        if (hours != 0) {
            $("#txtHour").setValue(hours);
        }
    }

    function PatientPopUp(e) {
        if (e.keyCode == 13) {
            var keyWord = $("#txtSearchPatient").getValue();
            if (keyWord != "" || keyWord != undefined) {
                $("#loading-image").show();
                var _url = '/Common/GetPatientBySearch';
                //var _url = '@Url.Action("GetPatientBySearch", "Common")';
                var data = JSON.stringify({ keyWord: keyWord });
                $.AjaxJsonRequest(_url, data, function (msg) {

                    var lstPatient = msg.lstPatient;
                    if (lstPatient != null) {
                        $("#modalboxPatient").show("modal");
                        $('#tblPopUpPatient').dataTable().fnDestroy(); // Destroy DataTables from current table

                    else {
                        ShowMessage(1, "No Record Found");
                        $("#modalboxPatient").hide("modal");
                        $("#loading-image").hide();
                    }
                });
            }
        }
    }

    function ValidateSelectedSlots() {
        let selectedInquiryIds = lstSlotSaveArray.map(b => parseInt(b.InquiryScheduleSlotId, 10));
        let selectedSlots = selectedInquiryIds
            .map(id => lstSlots.find(a => a.InquiryScheduleSlotId === id)?.SlotNo)
            .filter(slot => slot !== undefined);

        // Sort the slot numbers
        selectedSlots.sort((a, b) => a - b);

        // Function to check if slots are consecutive
        function isConsecutive(arr) {
            for (let i = 1; i < arr.length; i++) {
                if (arr[i] !== arr[i - 1] + 1) {
                    return false;
                }
            }
            return true;
        }

        // Validate selection
        if (!isConsecutive(selectedSlots)) {
            ShowMessage(1, "You can only select consecutive slots.");
            return false;
        } else {
            return true;
        }
    }
